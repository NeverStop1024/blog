    实现页面缓存

    // 缓存在<router-view></router-view>位置处展示的所有组件
<keep-alive>
    <router-view></router-view>
</keep-alive>

// 缓存在<router-view></router-view>位置处展示的单个指定组件
<keep-alive include="组件名">
    <router-view></router-view>
</keep-alive>

// 缓存在<router-view></router-view>位置处展示的多个指定组件
<keep-alive include="['组件名1', '组件名2']">
    <router-view></router-view>
</keep-alive>
